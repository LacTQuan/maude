      Test the trivial case where the start and goal are the same.

      """
      grid = [
          [0, 0],
          [0, 0]
      ]
      start = Node(0, 0)

      start.id = 0
      start.pid = 0
      astar = AStar(grid)
      found, path = astar.plan(start, goal)
      assert found is True, "Path should be found when start equals goal"

      assert len(path) == 1

      Test a simple 5x5 grid with no obstacles.

      found, path = astar.plan(start, goal)
